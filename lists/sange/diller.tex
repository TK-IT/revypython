\begin{Sang}[Michael Jackson -- Thriller]{Diller!}

\begin{Persongalleri}
  \item Mand, der er til fest (A)
  \item Danser 1 (D1)
  \item Danser 2 (D2)
  \item Danser 3 (D3)
  \item Danser 4 (D4)
\end{Persongalleri}

\begin{Regi}
  A på scenen, forscene.
\end{Regi}

\begin{Strofe}
  \Vers[A]{Det' tæt på midnat,}%
  \Vers[---]{jeg føler mig for alvor i humør.}%
  \Vers[---]{Det er på tid' at}%
  \Vers[---]{jeg giver aft'nen lidt mere kulør.}%
  \Vers[---]{Mit underliv}%
  \Vers[---]{det siger, pigen hun er fa'me rar.}%
  \Vers[---]{Når hun er stiv,}%
  \Vers[---]{så låner jeg lig' hendes kamera -- og bill'der tar'!}%
\end{Strofe}

\begin{Regi}
  Her er der igen mulighed for at bruge tear-off-bukser, hvor A har et par tights indenunder med en ordentlig slange i. Tæppe hvor D1-4 afsløres. Folk har alle sammen en ordenlig slange mellem benene. Der danses.
\end{Regi}

\begin{Strofe}[Omk.]
\Vers[---]{Det er min diller}%
\Vers[---]{min dillertjavs}%
\Vers[---]{og på din mem'ry-stick}%
\Vers[---]{der kan du nu finde min pik!}%
\Vers[---]{Hils på min diller}%
\Vers[---]{min tissetrold}%
\Vers[---]{Luk op for din mobil}%
\Vers[---]{og se en vilder' diller -- lige he-e-er!}%
\end{Strofe}

\begin{Regi}
   Dans fortsætter cirka et omkvæd. Der bliver svunget slanger i vildskab. Tæppe på et passende tidspunkt.
\end{Regi}

\end{Sang}
